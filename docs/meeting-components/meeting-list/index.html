Example Usage​

import { MeetingList } from "react-chat-elements"

<MeetingList
  className='meeting-list'
  dataSource={[
    {
      id: '1',
      subject: 'Calling',
      date: new Date(),
      avatars: [{
          src: 'https://avatars.githubusercontent.com/u/80540635?v=4',
      }]
    },
    {
      id: '1',
      subject: 'Calling',
      date: new Date(),
      avatars: [
        {
          src: 'https://avatars.githubusercontent.com/u/41473129?v=4',
        },
      ]
    },
  ]}
/>
